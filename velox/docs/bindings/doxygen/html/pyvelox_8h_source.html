48inline void addVeloxBindings(pybind11::module& m, bool asLocalModule = true) {
49 // Inlining these bindings since adding them to the cpp file results in a
50 // ASAN error.
51 using namespace velox;
52 namespace py = pybind11;
53
54 // Add TypeKind enum.
55 py::enum_<velox::TypeKind>(m, "TypeKind", py::module_local(asLocalModule))
56 .value("BOOLEAN", velox::TypeKind::BOOLEAN)
57 .value("TINYINT", velox::TypeKind::TINYINT)
58 .value("SMALLINT", velox::TypeKind::SMALLINT)
59 .value("INTEGER", velox::TypeKind::INTEGER)
60 .value("BIGINT", velox::TypeKind::BIGINT)
61 .value("REAL", velox::TypeKind::REAL)
62 .value("DOUBLE", velox::TypeKind::DOUBLE)
63 .value("VARCHAR", velox::TypeKind::VARCHAR)
64 .value("VARBINARY", velox::TypeKind::VARBINARY)
65 .value("TIMESTAMP", velox::TypeKind::TIMESTAMP)
66 .value("OPAQUE", velox::TypeKind::OPAQUE)
67 .value("ARRAY", velox::TypeKind::ARRAY)
68 .value("MAP", velox::TypeKind::MAP)
69 .value("ROW", velox::TypeKind::ROW)
70 .export_values();
71
72 // Create VeloxType bound to velox::Type.
73 py::class_<Type, std::shared_ptr<Type>> type(
74 m, "VeloxType", py::module_local(asLocalModule));
75
76 // Adding all the derived types of Type here.
77 py::class_<BooleanType, Type, std::shared_ptr<BooleanType>> booleanType(
78 m, "BooleanType", py::module_local(asLocalModule));
79 py::class_<IntegerType, Type, std::shared_ptr<IntegerType>> integerType(
80 m, "IntegerType", py::module_local(asLocalModule));
81 py::class_<BigintType, Type, std::shared_ptr<BigintType>> bigintType(
82 m, "BigintType", py::module_local(asLocalModule));
83 py::class_<SmallintType, Type, std::shared_ptr<SmallintType>> smallintType(
84 m, "SmallintType", py::module_local(asLocalModule));
85 py::class_<TinyintType, Type, std::shared_ptr<TinyintType>> tinyintType(
86 m, "TinyintType", py::module_local(asLocalModule));
87 py::class_<RealType, Type, std::shared_ptr<RealType>> realType(
88 m, "RealType", py::module_local(asLocalModule));
89 py::class_<DoubleType, Type, std::shared_ptr<DoubleType>> doubleType(
90 m, "DoubleType", py::module_local(asLocalModule));
91 py::class_<TimestampType, Type, std::shared_ptr<TimestampType>> timestampType(
92 m, "TimestampType", py::module_local(asLocalModule));
93 py::class_<VarcharType, Type, std::shared_ptr<VarcharType>> varcharType(
94 m, "VarcharType", py::module_local(asLocalModule));
95 py::class_<VarbinaryType, Type, std::shared_ptr<VarbinaryType>> varbinaryType(
96 m, "VarbinaryType", py::module_local(asLocalModule));
97 py::class_<ArrayType, Type, std::shared_ptr<ArrayType>> arrayType(
98 m, "ArrayType", py::module_local(asLocalModule));
99 py::class_<MapType, Type, std::shared_ptr<MapType>> mapType(
100 m, "MapType", py::module_local(asLocalModule));
101 py::class_<RowType, Type, std::shared_ptr<RowType>> rowType(
102 m, "RowType", py::module_local(asLocalModule));
103 py::class_<FixedSizeArrayType, Type, std::shared_ptr<FixedSizeArrayType>>
104 fixedArrayType(m, "FixedSizeArrayType", py::module_local(asLocalModule));
105
106 // Basic operations on Type.
107 type.def("__str__", &Type::toString);
108 // Gcc doesnt support the below kind of templatization.
109#if defined(__clang__)
110 // Adds equality and inequality comparison operators.
111 type.def(py::self == py::self);
112 type.def(py::self != py::self);
113#endif
114 type.def(
115 "cpp_size_in_bytes",
116 &Type::cppSizeInBytes,
117 "Return the C++ size in bytes");
118 type.def(
119 "is_fixed_width",
120 &Type::isFixedWidth,
121 "Check if the type is fixed width");
122 type.def(
123 "is_primitive_type",
124 &Type::isPrimitiveType,
125 "Check if the type is a primitive type");
126 type.def("kind", &Type::kind, "Returns the kind of the type");
127 type.def("serialize", &serializeType, "Serializes the type as JSON");
128
129 booleanType.def(py::init());
130 tinyintType.def(py::init());
131 smallintType.def(py::init());
132 integerType.def(py::init());
133 bigintType.def(py::init());
134 realType.def(py::init());
135 doubleType.def(py::init());
136 varcharType.def(py::init());
137 varbinaryType.def(py::init());
138 timestampType.def(py::init());
139 arrayType.def(py::init<std::shared_ptr<Type>>());
140 arrayType.def(
141 "element_type", &ArrayType::elementType, "Return the element type");
142 fixedArrayType.def(py::init<int, velox::TypePtr>())
143 .def("element_type", &velox::FixedSizeArrayType::elementType)
144 .def("fixed_width", &velox::FixedSizeArrayType::fixedElementsWidth);
145 mapType.def(py::init<std::shared_ptr<Type>, std::shared_ptr<Type>>());
146 mapType.def("key_type", &MapType::keyType, "Return the key type");
147 mapType.def("value_type", &MapType::valueType, "Return the value type");
148
149 rowType.def(py::init<
150 std::vector<std::string>,
151 std::vector<std::shared_ptr<const Type>>>());
152 rowType.def("size", &RowType::size, "Return the number of columns");
153 rowType.def(
154 "child_at",
155 &RowType::childAt,
156 "Return the type of the column at a given index",
157 py::arg("idx"));
158 rowType.def(
159 "find_child",
160 [](const std::shared_ptr<RowType>& type, const std::string& name) {
161 return type->findChild(name);
162 },
163 "Return the type of the column with the given name",
164 py::arg("name"));
165 rowType.def(
166 "get_child_idx",
167 &RowType::getChildIdx,
168 "Return the index of the column with the given name",
169 py::arg("name"));
170 rowType.def(
171 "name_of",
172 &RowType::nameOf,
173 "Return the name of the column at the given index",
174 py::arg("idx"));
175 rowType.def("names", &RowType::names, "Return the names of the columns");
176}
